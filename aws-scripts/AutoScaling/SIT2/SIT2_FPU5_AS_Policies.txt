aws autoscaling put-scaling-policy --policy-name evise-fpu5-sit2-scaleout-policy --auto-scaling-group-name ASG --scaling-adjustment 1 --adjustment-type ChangeInCapacity 

#Pin CW Alarm to Autoscaling Policy to scale up 
aws cloudwatch put-metric-alarm --alarm-name AddCapacitySit2FPU5 --metric-name ApproximateNumberOfMessagesVisible --namespace "AWS/SQS" --statistic Average --period 300 --threshold 100 --comparison-operator GreaterThanOrEqualToThreshold --dimensions Name=QueueName,Value=sit2-event-queue-GenerateConsolidatedPDF --evaluation-periods 2 --alarm-actions arn
